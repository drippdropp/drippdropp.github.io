Variation on Selection Sort

This isn't a "true" selection sort technically. More of a "linear sort."

function minidx(ary)
    (idx, elem) = (1, ary[1])
    for (k, el) in enumerate(ary)
        if el < elem
            (idx, elem) = (k, el)
        end
    end
    return idx
end
