A Fibonacci Implementation

function fib(n)
	n ≤ zero(n) && return nothing
	n == one(n) && return [one(n)]

	reduce( (acc, el) -> vcat(acc, big(acc[el-1])+acc[el-2]) , 3:n; init=[1,1])
end
